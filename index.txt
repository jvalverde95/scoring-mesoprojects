/* ══════════════════════════════════════════════════
   FÓRMULA DEL ALGORITMO (del Excel Hoja3 / Hoja2)
   ══════════════════════════════════════════════════
   Base: e_custom = 2 (base logarítmica del modelo)

   Por cada subcriterio i con parámetros:
     C0, Beta0, Alfa0  → curva sigmoid base (tramo bajo)
     C,  Beta,  Alfa   → curva sigmoid boost (tramo alto)

   sigmoid_base(S)  = Alfa0 / (1 + 2^(-Beta0 * (S - C0)))
   sigmoid_boost(S) = Alfa  / (1 + 2^(-Beta  * (S - C )))
   Boost_i(S)       = max(sigmoid_base(S), sigmoid_boost(S), 1.0)

   Score dimensión  = Σ(P_i × Boost_i × S_i) / Σ(P_i × Boost_i)
   Score global     = Σ(peso_dim_j × Score_dim_j)  [ponderación lineal entre dims]
   ══════════════════════════════════════════════════ */

const E = 2; // base del modelo (no es e de Euler)

function sigmoidBoost(S, C0, Beta0, Alfa0, C, Beta, Alfa) {
  const base    = Alfa0 / (1 + Math.pow(E, -Beta0 * (S - C0)));
  const boosted = Alfa  / (1 + Math.pow(E, -Beta  * (S - C )));
  return Math.max(base, boosted, 1.0);
}

function scoreDim(criterios) {
  let num = 0, den = 0;
  criterios.forEach(c => {
    const b = sigmoidBoost(c.val, c.C0, c.Beta0, c.Alfa0, c.C, c.Beta, c.Alfa);
    num += c.pw * b * c.val;
    den += c.pw * b;
  });
  return den > 0 ? num / den : 0;
}

function scoreGlobal() {
  return DIMS.reduce((acc, d) => acc + d.peso * scoreDim(d.criterios), 0);
}

/* ── RENDER BOOST PREVIEW ────────────── */
function boostPreview(crit) {
  const maxB = 10;
  return Array.from({length:10},(_,i)=>{
    const s = i+1;
    const b = sigmoidBoost(s, crit.C0, crit.Beta0, crit.Alfa0, crit.C, crit.Beta, crit.Alfa);
    const h = Math.min(Math.round((b/maxB)*32), 32);
    const col = scColor(b);
    const active = s === crit.val ? 'border:1.5px solid #111;' : '';
    return `<div class="bp-col">
      <div class="bp-bar-wrap"><div class="bp-bar" style="height:${h}px;background:${col};opacity:${s===crit.val?1:0.4};${active}"></div></div>
      <div class="bp-lbl">${s}</div>
    </div>`;
  }).join('');
}

/* ── SET PARAM ───────────────────────── */

  if(isNaN(v)) return;
  c[param] = v;
  const boost = sigmoidBoost(c.val,c.C0,c.Beta0,c.Alfa0,c.C,c.Beta,c.Alfa);
  const bl = document.getElementById('cb-'+cid);
  if(bl) bl.textContent = 'boost: '+boost.toFixed(2)+'×';
  const bp = document.getElementById('bp-'+cid);
  if(bp) bp.innerHTML = boostPreview(c);
  updDimScore(did); upd();
}

function resetParams(cid, did) {
  const c = findCrit(cid);

  updDimScore(did); upd();
}

function findCrit(cid) {
  for(const d of DIMS) { const c=d.criterios.find(x=>x.id===cid); if(c)return c; }
  return null;
}

  if(d) d.peso = v/100;
  checkTotal(); upd();
}

  });
}

let portfolioData = [];
let portfolioSort = 'score';
let portfolioFilter = 'all';

// Column mapping: which Excel columns map to which criteria
// The Excel has: col A=Nº, col B=Nombre, col C onwards = criteria in order
// Criteria order matches DIMS order
const CRIT_ORDER = [];
DIMS.forEach(d => d.criterios.forEach(c => CRIT_ORDER.push(c.id)));

function loadExcel(input) {
  const file = input.files[0];
  if (!file) return;
  const reader = new FileReader();
  reader.onload = e => {
    try {
      const wb = XLSX.read(e.target.result, {type:'array'});
      // Try "Carga de Proyectos" sheet first, fallback to first sheet
      const sheetName = wb.SheetNames.includes('Carga de Proyectos')
        ? 'Carga de Proyectos' : wb.SheetNames[0];
      const ws = wb.Sheets[sheetName];
      const data = XLSX.utils.sheet_to_json(ws, {header:1, defval:''});

      // Find data rows: skip until we find the header row (row with "Nombre del proyecto")
      // Data starts at row index 5 (0-based: rows 0-4 are header, data from row 5)
      // Col 0=Nº, Col 1=Nombre, Col 2..N=criteria, then score/cls cols (we ignore those, recalc)
      const projects = [];
      for (let r = 5; r < data.length; r++) {
        const row = data[r];
        const nom = String(row[1] || '').trim();
        if (!nom) continue;
        const area = String(row[2] || '').trim(); // might be area in some versions
        // Extract scores: cols 2 onwards (or 3 if area is col 2)
        // Detect: check if col 2 is a number (score) or text (area)
        let scoreStart = 2;
        if (typeof row[2] === 'string' && isNaN(parseFloat(row[2]))) scoreStart = 3;
        // Actually in our Excel: col B=nombre, col C=area dropdown... wait
        // Our Excel: A=Nº, B=Nombre+Área (same cell with dropdown), C onwards = criteria
        // Let's be flexible: scan for first numeric col >= col 2
        let firstNumCol = -1;
        for (let c = 2; c < row.length; c++) {
          const v = parseFloat(row[c]);
          if (!isNaN(v) && v >= 1 && v <= 10) { firstNumCol = c; break; }
        }
        if (firstNumCol === -1) continue;

        const scores = {};
        CRIT_ORDER.forEach((cid, idx) => {
          const v = parseFloat(row[firstNumCol + idx]);
          scores[cid] = (!isNaN(v) && v >= 1 && v <= 10) ? v : 5;
        });
        projects.push({ nom, area: nom, scores });
      }

      if (projects.length === 0) {
        toast('No se encontraron proyectos válidos en el Excel');
        return;
      }

      // Compute scores using our algorithm
      portfolioData = projects.map(proj => {
        // Temporarily assign values to compute
        const dimScores = DIMS.map(d => {
          let num = 0, den = 0;
          d.criterios.forEach(c => {
            const s = proj.scores[c.id] || 5;
            const b = sigmoidBoost(s, c.C0, c.Beta0, c.Alfa0, c.C, c.Beta, c.Alfa);
            num += c.pw * b * s;
            den += c.pw * b;
          });
          return den > 0 ? num/den : 0;
        });
        const finalScore = DIMS.reduce((acc,d,i) => acc + d.peso * dimScores[i], 0);
        const d1Score = dimScores[0];
        return { ...proj, dimScores, finalScore, d1Score, autoP: d1Score >= 8 };
      });

      renderPortfolio();
      document.getElementById('portfolio-wrap').style.display = 'block';
      document.getElementById('clear-portfolio-btn').style.display = 'flex';
      document.getElementById('import-info').textContent =
        `${portfolioData.length} proyectos cargados desde "${file.name}"`;
      toast(`✓ ${portfolioData.length} proyectos importados correctamente`);
    } catch(err) {
      console.error(err);
      toast('Error al leer el Excel. Verifica que sea el archivo de carga de mesoestetic.');
    }
    input.value = '';
  };
  reader.readAsArrayBuffer(file);
}

function computePortfolioScore(proj) {
  // Use current DIMS params (respects any param changes made by user)
  const dimScores = DIMS.map(d => {
    let num = 0, den = 0;
    d.criterios.forEach(c => {
      const s = proj.scores[c.id] || 5;
      const b = sigmoidBoost(s, c.C0, c.Beta0, c.Alfa0, c.C, c.Beta, c.Alfa);
      num += c.pw * b * s;
      den += c.pw * b;
    });
    return den > 0 ? num/den : 0;
  });
  const finalScore = DIMS.reduce((acc,d,i) => acc + d.peso * dimScores[i], 0);
  return { dimScores, finalScore, d1Score: dimScores[0] };
}

function sortPortfolio(by) {
  portfolioSort = by;
  renderPortfolio();
}

function filterPortfolio(f, btn) {
  portfolioFilter = f;
  document.querySelectorAll('.port-filter-btn').forEach(b => b.classList.remove('active'));
  if (btn) btn.classList.add('active');
  renderPortfolio();
}

function renderPortfolio() {
  // Re-compute with current params
  portfolioData.forEach(proj => {
    const r = computePortfolioScore(proj);
    proj.dimScores = r.dimScores;
    proj.finalScore = r.finalScore;
    proj.d1Score = r.d1Score;
    proj.autoP = r.d1Score >= 8;
  });

  // Sort
  const sorted = [...portfolioData].sort((a,b) => {
    if (portfolioSort === 'score') return b.finalScore - a.finalScore;
    return a.nom.localeCompare(b.nom);
  });

  const tbody = document.getElementById('port-tbody');
  tbody.innerHTML = '';
  let visCount = 0;

  sorted.forEach((proj, idx) => {
    const cl = clsf(proj.finalScore);
    const hide = portfolioFilter !== 'all' && !cl.et.includes(portfolioFilter);
    if (!hide) visCount++;

    const tr = document.createElement('tr');
    tr.className = hide ? 'hidden' : '';
    tr.dataset.cls = cl.et;

    const dimCells = proj.dimScores.map((ds, di) => {
      const col = scColor(ds);
      return `<td style="text-align:center;color:${col};font-weight:600;font-size:11px;">${ds.toFixed(1)}</td>`;
    }).join('');

    const clsStyle = proj.autoP
      ? `background:#FBF0F0;color:#8B2020;border-color:#D4A0A0;`
      : `background:${cl.bg};color:${cl.c};border-color:${cl.b};`;

    tr.innerHTML = `
      <td><span class="rank-badge">${idx+1}</span></td>
      <td style="font-weight:600;max-width:220px;white-space:nowrap;overflow:hidden;text-overflow:ellipsis;" title="${proj.nom}">${proj.nom}</td>
      <td style="font-size:10px;color:#AAA;">${proj.area || '—'}</td>
      ${dimCells}
      <td style="text-align:center;">
        <span class="port-score" style="color:${scColor(proj.finalScore)};">${proj.finalScore.toFixed(1)}</span>
      </td>
      <td><span class="port-cls" style="${clsStyle}">${proj.autoP ? '⚑ AUTO-PRIORITARIO' : cl.et}</span></td>
      <td class="port-d1" style="text-align:center;">${proj.autoP ? '⚑ Sí' : '—'}</td>
      <td><button class="port-load-btn" onclick="loadProjectIntoEval(${portfolioData.indexOf(proj)})">Evaluar →</button></td>
    `;
    tbody.appendChild(tr);
  });

  document.getElementById('portfolio-count').textContent =
    `${visCount} de ${portfolioData.length} proyectos`;
  document.getElementById('port-empty').style.display = visCount === 0 ? 'block' : 'none';
  document.getElementById('port-tbl').style.display = visCount === 0 ? 'none' : 'table';
}

function loadProjectIntoEval(idx) {
  const proj = portfolioData[idx];
  if (!proj) return;
  // Set project name
  document.getElementById('f-name').value = proj.nom;
  // Set scores in sliders
  DIMS.forEach(d => d.criterios.forEach(c => {
    const v = proj.scores[c.id] || 5;
    c.val = v;
    const sl = document.getElementById('sl-'+c.id);
    const cv = document.getElementById('cv-'+c.id);
    const cb = document.getElementById('cb-'+c.id);
    const bp = document.getElementById('bp-'+c.id);
    if (sl) sl.value = v;
    if (cv) { cv.textContent = v; cv.style.color = scColor(v); }
    if (cb) cb.textContent = 'boost: '+sigmoidBoost(v,c.C0,c.Beta0,c.Alfa0,c.C,c.Beta,c.Alfa).toFixed(2)+'×';
    if (bp) bp.innerHTML = boostPreview(c);
  }));
  upd();
  // Scroll to evaluator
  document.getElementById('dims-col').scrollIntoView({behavior:'smooth', block:'start'});
  toast(`Proyecto "${proj.nom}" cargado en el evaluador`);
}
